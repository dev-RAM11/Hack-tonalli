Ejemplo: Integrar pago de suscripción en XLM

Este es el punto de integración más directo. Actualmente upgradePlan() cambia el plan sin cobrar. Así se conectaría con Stellar:

// ── stellar.service.ts — Nuevo método ──

async chargeSubscription(userSecret: string, amountUsd: number): Promise<{success, txHash}> {
  // 1. Obtener precio XLM actual (API o fijo para demo)
  const xlmPrice = 0.15; // USD por XLM
  const xlmAmount = (amountUsd / xlmPrice).toFixed(2);

  // 2. Crear transacción: usuario → pool de Tonalli
  const userKeypair = Keypair.fromSecret(userSecret);
  const account = await server.loadAccount(userKeypair.publicKey());
  const tx = new TransactionBuilder(account, { fee: '100' })
    .addOperation(Operation.payment({
      destination: TONALLI_POOL_PUBLIC_KEY,
      asset: Asset.native(),
      amount: xlmAmount,
    }))
    .addMemo(Memo.text(`sub:${plan}`))
    .setTimeout(30).build();

  // 3. Firmar y enviar
  tx.sign(userKeypair);
  const result = await server.submitTransaction(tx);
  return { success: true, txHash: result.hash };
}
// ── users.service.ts — Modificar upgradePlan() ──

async upgradePlan(userId: string, plan: 'pro' | 'max'): Promise<User> {
  const user = await this.findById(userId);
  const prices = { pro: 3, max: 8 };

  // Cobrar en XLM si tiene wallet
  if (user.stellarSecretKey) {
    const result = await this.stellarService.chargeSubscription(
      user.stellarSecretKey, prices[plan]
    );
    if (!result.success) throw new Error('Pago fallido');
    // Guardar txHash como comprobante
    user.subscriptionTxHash = result.txHash;
  }

  user.plan = plan;
  return this.userRepository.save(user);
}
